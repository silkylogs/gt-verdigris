cmake_minimum_required(VERSION 3.20)
project(
	green_top_verdigris
	VERSION 1.0
	LANGUAGES CXX
)

add_library(forth_lib STATIC src/forth/forth.hpp src/forth/forth.cpp)
target_include_directories(forth_lib PUBLIC src/forth)
target_compile_features(forth_lib PUBLIC cxx_std_20)
set_target_properties(forth_lib PROPERTIES LINKER_LANGUAGE CXX)

add_executable(game_executable src/main.cpp)
target_compile_options(game_executable PRIVATE
     $<$<OR:$<CXX_COMPILER_ID:Clang>,$<CXX_COMPILER_ID:AppleClang>,$<CXX_COMPILER_ID:GNU>>:
	-Wall -Werror -Wextra -g3 -Wconversion -Wdouble-promotion
	-Wno-unused-parameter -Wno-unused-function -Wno-sign-conversion
	-fsanitize=undefined -fsanitize-trap>
     $<$<CXX_COMPILER_ID:MSVC>:
	/WX /W4 /Zi /RTCsu /Zc:__cplusplus>
)
target_compile_features(game_executable PUBLIC cxx_std_20)
target_link_libraries(game_executable PUBLIC forth_lib)
